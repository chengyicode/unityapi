Mathf.IsPowerOfTwo

public static bool IsPowerOfTwo(int value);

Description 描述

Returns true if the value is power of two.

using UnityEngine;

public class Example : MonoBehaviour
{
    void Start()
    {
        // prints false
        Debug.Log(Mathf.IsPowerOfTwo(7));

        // prints true
        Debug.Log(Mathf.IsPowerOfTwo(32));
    }
}
