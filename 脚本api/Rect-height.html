Rect.height

public float height;

Description 描述

The height of the rectangle, measured from the Y position.

Setting this value will also change yMax.

using UnityEngine;

public class Example : MonoBehaviour
{
    void Start()
    {
        Rect rect = new Rect(0, 0, 10, 10);
        print(rect.height);
        rect.height = 20;
    }
}
