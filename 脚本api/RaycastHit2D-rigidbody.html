RaycastHit2D.rigidbody

public Rigidbody2D rigidbody;

Description 描述

The Rigidbody2D attached to the object that was hit.

using UnityEngine;

public class Example : MonoBehaviour
{
    // Apply a force to a clicked rigidbody2D object.

    // The force applied to an object when hit.
    float hitForce;

    void Update()
    {
        //If the left mouse button is clicked.
        if (Input.GetMouseButtonDown(0))
        {
            //Get the mouse position on the screen and send a raycast into the game world from that position.
            Vector2 worldPoint = Camera.main.ScreenToWorldPoint(Input.mousePosition);

            RaycastHit2D hit = Physics2D.Raycast(worldPoint, Vector2.zero);

            if (hit.rigidbody != null)
            {
                hit.rigidbody.AddForce(Vector2.one * hitForce);
            }
        }
    }
}
