Gyroscope.enabled

public bool enabled;

Description 描述

Sets or retrieves the enabled status of this gyroscope.

using UnityEngine;

public class Example : MonoBehaviour
{
    bool enableTilt;

    void OnGUI()
    {
        if (Input.gyro.enabled)
        {
            GUILayout.Toggle(enableTilt, "Enable tilt control");
        }

        // Other GUI elements...
    }
}
