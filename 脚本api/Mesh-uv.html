public Vector2[] uv;

Description 描述

The base texture coordinates of the Mesh.

using UnityEngine;
using System.Collections;

public class ExampleClass : MonoBehaviour
{
    void Start()
    {
        Mesh mesh = GetComponent<MeshFilter>().mesh;
        Vector3[] vertices = mesh.vertices;
        Vector2[] uvs = new Vector2[vertices.Length];

        for (int i = 0; i < uvs.Length; i++)
        {
            uvs[i] = new Vector2(vertices[i].x, vertices[i].z);
        }
        mesh.uv = uvs;
    }
}

Note: Texture coordinates must be created or modified externally from the mesh. The script example above creates the uvs array, assigns texture coordinates to the uvs array, and then assigns the array to the Mesh.
