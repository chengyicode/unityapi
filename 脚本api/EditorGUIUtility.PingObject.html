EditorGUIUtility.PingObject

public static void PingObject(Object obj);
public static void PingObject(int targetInstanceID);

Parameters

objThe object to be pinged.

Description 描述

Ping an object in the Scene like clicking it in an inspector.

PingObject will cause the Hierarchy to highlight the pinged object. The pinged object does not have to selected. For example GameObject.Find can be used to locate an object to ping.

// Pings the currently selected Object
using UnityEditor;
using UnityEngine;

public class PingObjectExample
{
    [MenuItem("Examples/Ping Selected")]
    static void Ping()
    {
        if (!Selection.activeObject)
        {
            Debug.Log("Select an object to ping");
            return;
        }

        EditorGUIUtility.PingObject(Selection.activeObject);
    }
}
