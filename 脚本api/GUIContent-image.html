using UnityEngine;

public class ExampleScript : MonoBehaviour
{
    public Texture icon;

    void OnGUI()
    {
        if (!icon)
        {
            Debug.LogError("Add a texture on the inspector first");
            return;
        }
        GUI.Button(new Rect(0, 0, 100, 20), new GUIContent(icon));
    }
}
