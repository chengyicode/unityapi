Logger.LogWarning

public void LogWarning(string tag, object message);
public void LogWarning(string tag, object message, Object context);

Parameters

tagUsed to identify the source of a log message. It usually identifies the class where the log call occurs.
messageString or object to be converted to string representation for display.
contextObject to which the message applies.

Description 描述

A variant of Logger.Log that logs an warning message.

using UnityEngine;
using System.Collections;

public class MyGameClass : MonoBehaviour
{
    private static ILogger logger = Debug.unityLogger;
    private static string kTAG = "MyGameTag";
    private Transform tr;

    void MyGameMethod()
    {
        if (tr == null)
        {
            logger.LogWarning(kTAG, "memberVariable must be set to point to a Transform.", tr);

            Debug.unityLogger.LogWarning(kTAG, "Failed! to set Transform",  tr);
        }
    }
}
