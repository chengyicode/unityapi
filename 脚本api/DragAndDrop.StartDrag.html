DragAndDrop.StartDrag

public static void StartDrag(string title);

Description 描述

Start a drag operation.

Initiates a drag operation with the current drag object state. Use paths and/or objectReferences to setup drag state.

See Also: PrepareStartDrag, paths, objectReferences.

using UnityEngine;
using UnityEditor;

public class Example : MonoBehaviour
{
    void OnGUI()
    {
        if (Event.current.type == EventType.MouseDrag)
        {
            // Clear out drag data
            DragAndDrop.PrepareStartDrag();

            // Set up what we want to drag
            DragAndDrop.paths[0] = "/Users/joe/myPath.txt";

            // Start the actual drag
            DragAndDrop.StartDrag("Dragging title");

            // Make sure no one uses the event after us
            Event.current.Use();
        }
    }
}
