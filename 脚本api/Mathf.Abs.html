Mathf.Abs

public static float Abs(float f);

Description 描述

Returns the absolute value of f.

using UnityEngine;

public class ScriptExample : MonoBehaviour
{
    void Start()
    {
        // prints 10.5
        Debug.Log(Mathf.Abs(-10.5f));
    }
}
