Color32.Lerp

public static Color32 Lerp(Color32 a, Color32 b, float t);

Description 描述

Linearly interpolates between colors a and b by t.

t is clamped between 0 and 1. When t is 0 returns a. When t is 1 returns b.

using UnityEngine;

public class Example : MonoBehaviour
{
    Color32 lerpedColor = Color.white;
    Color32 white32 = Color.white;
    Color32 black32 = Color.black;

    void Update()
    {
        lerpedColor = Color32.Lerp(white32, black32, Time.time);
    }
}
