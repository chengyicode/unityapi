ParticleSystem.TrailModule.inheritParticleColor

Switch to Manual
public bool inheritParticleColor;

Description 描述

Toggle whether the trail will inherit the particle color as its starting color.

Any Trail module curves will be applied on top of the particle color, if enabled.

using UnityEngine;
using System.Collections;

public class ExampleClass : MonoBehaviour
{
    private ParticleSystem ps;
    public bool inheritParticleColor = true;

    void Start()
    {
        ps = GetComponent<ParticleSystem>();

        var main = ps.main;
        main.startColor = new ParticleSystem.MinMaxGradient(Color.red, Color.yellow);
        main.startSize = 0.1f;

        var trails = ps.trails;
        trails.enabled = true;

        var psr = GetComponent<ParticleSystemRenderer>();
        psr.trailMaterial = new Material(Shader.Find("Sprites/Default"));
    }

    void Update()
    {
        var trails = ps.trails;
        trails.inheritParticleColor = inheritParticleColor;
    }

    void OnGUI()
    {
        inheritParticleColor = GUI.Toggle(new Rect(25, 25, 200, 30), inheritParticleColor, "Inherit Particle Color");
    }
}
