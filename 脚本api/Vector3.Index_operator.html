public float this[int];

Description 描述

Access the x, y, z components using [0], [1], [2] respectively.

using UnityEngine;
using System.Collections;

public class ExampleClass : MonoBehaviour
{
    public Vector3 p;

        p[1] = 5.0f;
    }
}
