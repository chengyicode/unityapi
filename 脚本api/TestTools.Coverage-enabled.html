Coverage.enabled

public static bool enabled;

Description 描述

This is a read only property that returns whether coverage is enabled.

using UnityEngine;
using UnityEngine.TestTools;

public class CoverageExample : MonoBehaviour
{
    void Start()
    {
        if (Coverage.enabled)
            Debug.Log("Coverage is enabled.");
        else
            Debug.Log("Coverage is not enabled.");
    }
}
