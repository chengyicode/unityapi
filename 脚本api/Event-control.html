Event.control

public bool control;

Description 描述

Is Control key held down? (Read Only)

Returns true if any Control key is held down.

using UnityEngine;

public class Example : MonoBehaviour
{
    void OnGUI()
    {
        Event e = Event.current;
        if (e.control)
        {
            Debug.Log("Control was pressed.");
        }
    }
}
