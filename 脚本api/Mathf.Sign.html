Mathf.Sign

public static float Sign(float f);

Description 描述

Returns the sign of f.

Return value is 1 when f is positive or zero, -1 when f is negative.

using UnityEngine;

public class Example : MonoBehaviour
{
    void Start()
    {
        Debug.Log(Mathf.Sign(-10));
        Debug.Log(Mathf.Sign(10));
    }
}
