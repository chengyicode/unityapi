Event.isKey

public bool isKey;

Description 描述

Is this event a keyboard event? (Read Only)

using UnityEngine;

public class Example : MonoBehaviour
{
    // Detects any keyboard event
    void OnGUI()
    {
        Event e = Event.current;
        if (e.isKey)
        {
            Debug.Log("Detected a keyboard event!");
        }
    }
}
