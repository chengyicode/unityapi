KeyCode.UpArrow

Description 描述

Up arrow key.

Use this as a parameter to a function like Input.GetKey to detect when the user presses the up arrow key. See Also: Input.GetKey, Input.GetKeyDown, Input.GetKeyUp.

//Attach this to a GameObject
//This script tells when the up arrow key is pressed down and when it is released
using UnityEngine;

public class Example : MonoBehaviour
{
    void Update()
    {
        //Detect when the up arrow key is pressed down
        if (Input.GetKeyDown(KeyCode.UpArrow))
            Debug.Log("Up Arrow key was pressed.");

        //Detect when the up arrow key has been released
        if (Input.GetKeyUp(KeyCode.UpArrow))
            Debug.Log("Up Arrow key was released.");
    }
}
