Material.SetTextureScale

Switch to Manual
public void SetTextureScale(string name, Vector2 value);
public void SetTextureScale(int nameID, Vector2 value);

Parameters

nameIDProperty name ID, use Shader.PropertyToID to get it.
nameProperty name, e.g. "_MainTex".
valueTexture placement scale.

Description 描述

Sets the placement scale of texture propertyName.

See Also: mainTextureScale property, GetTextureScale, SetTexture.

using UnityEngine;

public class Example : MonoBehaviour
{
    // Scroll main texture based on time

    float scrollSpeed = 0.5f;
    Renderer rend;

    void Start()
    {
        rend = GetComponent<Renderer> ();
    }

    void Update()
    {
        // Animates main texture scale in a funky way!
        float scaleX = Mathf.Cos(Time.time) * 0.5f + 1;
        float scaleY = Mathf.Sin(Time.time) * 0.5f + 1;
        rend.material.SetTextureScale("_MainTex", new Vector2(scaleX, scaleY));
    }
}
