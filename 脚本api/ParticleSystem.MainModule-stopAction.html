ParticleSystem.MainModule.stopAction

Switch to Manual
public ParticleSystemStopAction stopAction;

Description 描述

Select whether to Disable or Destroy the GameObject, or to call the OnParticleSystemStopped script Callback, when the Particle System is stopped and all particles have died.

using UnityEngine;
using System.Collections;

public class ExampleClass : MonoBehaviour
{
    private ParticleSystem ps;

    void Start()
    {
        ps = GetComponent<ParticleSystem>();
        ps.Stop();

        var main = ps.main;
        main.loop = false;
        main.duration = 1.0f;
        main.stopAction = ParticleSystemStopAction.Destroy;

        ps.Play();
    }
}
