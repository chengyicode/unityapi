ParticleSystem.MainModule.startRotationZ

Switch to Manual
public ParticleSystem.MinMaxCurve startRotationZ;

Description 描述

The initial rotation of particles around the Z axis when emitted.

Note that the value should be given in radians. See Also: MinMaxCurve.

using UnityEngine;
using System.Collections;

public class ExampleClass : MonoBehaviour
{
    private ParticleSystem ps;
    public float hSliderValue = 1.0F;

    void Start()
    {
        ps = GetComponent<ParticleSystem>();

        var main = ps.main;
        main.startRotation3D = true;

        var psr = GetComponent<ParticleSystemRenderer>();
        psr.material = new Material(Shader.Find("Sprites/Default"));    // this material renders a square billboard, so we can see the rotation
    }

    void Update()
    {
        var main = ps.main;
        main.startRotationZ = hSliderValue;
    }

    void OnGUI()
    {
        hSliderValue = GUI.HorizontalSlider(new Rect(25, 45, 100, 30), hSliderValue, 0.0F, 360.0F * Mathf.Deg2Rad);
    }
}
