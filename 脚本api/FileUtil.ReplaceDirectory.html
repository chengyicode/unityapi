FileUtil.ReplaceDirectory

public static void ReplaceDirectory(string src, string dst);

Description 描述

Replaces a directory.

Replaces the directory located at dst with the directory located at src. if dst doesnt exists it copies the file. If dst exists then it deletes it and copies the directory at src to dst See Also: FileUtil.ReplaceFile.

using UnityEngine;
using UnityEditor;

public class ReplaceDir : MonoBehaviour
{
    [MenuItem("Example/Replace Directory")]
    static void ReplaceDirectory()
    {
        FileUtil.ReplaceDirectory("sourcepath/YourFolder", "destpath/YourNewFolder");
    }
}
