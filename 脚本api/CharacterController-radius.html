public float radius;

Description 描述

The radius of the character's capsule.

using UnityEngine;

public class Example : MonoBehaviour
{
    // Set the controller radius to 0.3f
    CharacterController controller;

    void Start()
    {
        controller = GetComponent<CharacterController>();
        controller.radius = 0.3f;
    }
}
