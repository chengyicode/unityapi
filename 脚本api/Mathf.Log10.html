Mathf.Log10

public static float Log10(float f);

Description 描述

Returns the base 10 logarithm of a specified number.

using UnityEngine;

public class Example : MonoBehaviour
{
    void Start()
    {
        // logarithm of 100 in base 10
        // Prints 2
        print(Mathf.Log10(100));
    }
}
