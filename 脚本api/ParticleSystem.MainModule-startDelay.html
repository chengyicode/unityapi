ParticleSystem.MainModule.startDelay

Switch to Manual
public ParticleSystem.MinMaxCurve startDelay;

Description 描述

Start delay in seconds.

Use this to delay when the playback starts on the system. See Also: MinMaxCurve.

using UnityEngine;
using System.Collections;

public class ExampleClass : MonoBehaviour
{
    private ParticleSystem ps;
    private bool restart;
    public float hSliderValue = 0.0F;

    void Start()
    {
        ps = GetComponent<ParticleSystem>();
    }

    void Update()
    {
        if (restart)
        {
            ps.Stop();
            ps.Clear();

            var main = ps.main;
            main.startDelay = hSliderValue;

            ps.Play();

            restart = false;
        }
    }

    void OnGUI()
    {
        hSliderValue = GUI.HorizontalSlider(new Rect(25, 45, 100, 30), hSliderValue, 0.0F, 5.0F);
        restart = GUI.Button(new Rect(25, 75, 100, 30), "Restart");
    }
}
