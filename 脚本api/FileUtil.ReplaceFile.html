FileUtil.ReplaceFile

public static void ReplaceFile(string src, string dst);

Description 描述

Replaces a file.

Replaces the file located at dst with the file located at src. if dst doesnt exists it just copies the file. If dst exists then it deletes it and copies the file at src to dst See Also: FileUtil.ReplaceDirectory.

using UnityEngine;
using UnityEditor;

public class ReplaceFile : MonoBehaviour
{
    [MenuItem("Example/Replace Something")]
    static void ReplaceSomething()
    {
        FileUtil.ReplaceFile("sourcepath/YourFileOrFolder", "destpath/YourReplacedFileOrFolder");
    }
}
