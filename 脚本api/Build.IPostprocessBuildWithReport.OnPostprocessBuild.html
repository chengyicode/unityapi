IPostprocessBuildWithReport.OnPostprocessBuild

public void OnPostprocessBuild(Build.Reporting.BuildReport report);

Parameters

reportA BuildReport containing information about the build, such as the target platform and output path.

Description 描述

Implement this function to receive a callback after the build is complete.

using UnityEditor;
using UnityEditor.Build;
using UnityEditor.Build.Reporting;
using UnityEngine;

class MyCustomBuildProcessor : IPostprocessBuildWithReport
{
    public int callbackOrder { get { return 0; } }
    public void OnPostprocessBuild(BuildReport report)
    {
        Debug.Log("MyCustomBuildProcessor.OnPostprocessBuild for target " + report.summary.platform + " at path " + report.summary.outputPath);
    }
}
