using UnityEngine;
using UnityEngine.Assertions;

public class AssertionExampleClass : MonoBehaviour
{
    MyClass myClassReference;

    void Update()
    {
        // Make sure the myClassReference reference is never set
        Assert.IsNull(myClassReference);
    }
}
