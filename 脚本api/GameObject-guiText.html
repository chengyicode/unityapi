Obsolete public Component guiText;

Description 描述

The GUIText attached to this GameObject (Read Only). (Null if there is none attached).

Obsolete. Use GetComponent<GUIText>() instead.

using UnityEngine;

public class Example : MonoBehaviour
{
    void Start()
    {
        GetComponent<GUIText>().text = "HelloWorld";
    }
}
